6) Pseudo-código\\
\newline

\begin{algorithm}[H]
\SetAlgoLined
\KwIn{SE, SD, CV}
\KwOut{S}

\If{SE = verdadeiro \textbf{ou} SD = verdadeiro \textbf{ou} CV = verdadeiro}{
    S = Recolher\;
}
\Else{
    S = Continuar\;
}

\caption{Algoritmo FODA}
\end{algorithm}
